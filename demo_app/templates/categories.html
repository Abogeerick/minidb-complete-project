-- Query using LEFT JOIN and GROUP BY:
SELECT c.id, c.name, c.color, c.icon, COUNT(e.id) as num_expenses
FROM categories c
LEFT JOIN expenses e ON c.id = e.category_id
GROUP BY c.id, c.name, c.color, c.icon
ORDER BY c.name;
