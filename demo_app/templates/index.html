-- Aggregate query for totals
SELECT SUM(amount) AS total, COUNT(*) AS cnt FROM expenses;
-- JOIN with GROUP BY for category breakdown
SELECT c.name, c.color, SUM(e.amount) AS total
FROM expenses e JOIN categories c ON e.category_id = c.id
GROUP BY c.name, c.color ORDER BY total DESC;
